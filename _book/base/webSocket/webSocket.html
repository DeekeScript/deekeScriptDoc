WebSocket вҖӢ

еҲӣе»әWebSocket вҖӢ

дҪҝз”ЁWebSocketзҡ„ж—¶еҖҷпјҢиҜ·еңЁдёҚдҪҝз”Ёзҡ„ж—¶еҖҷжүӢеҠЁе…ій—ӯпјӣеҰӮжһңжІЎжңүжүӢеҠЁе…ій—ӯпјҢеҪ“дҪҝз”ЁEngines.closeAll()зҡ„ж—¶еҖҷпјҢдјҡе…ій—ӯеҪ“еүҚзәҝзЁӢе’ҢеӯҗзәҝзЁӢзҡ„WebSocketгҖӮ

javascript
let client = new WebSocket("ws://192.168.0.100:8080");
client.onMessage = function (msg){
 	console.log('ж¶ҲжҒҜжқҘдәҶ' + msg); 
}

//иҝһжҺҘе®ҢжҲҗеҗҺ
client.onOpen = function (){
  client.send("жөӢиҜ•");//еҸ‘йҖҒж¶ҲжҒҜ
  setTimeout(() => {
    client.close();
  }, 1000);
}

//иҝһжҺҘе…ій—ӯжү§иЎҢ
client.onClose = function (code, reason){
    console.log('е…ій—ӯдәҶ',code, reason);
}

//еҮәзҺ°й”ҷиҜҜж—¶жү§иЎҢ
client.onError = function (errorMsg){
    console.log('еҮәй”ҷдәҶпјҡ' + errorMsg);
}

//дҝқжҢҒеҪ“еүҚзәҝзЁӢдёҚе…ій—ӯпјҢеҗҰеҲҷеҪ“еүҚдё»зәҝзЁӢдјҡз«Ӣй©¬е…ій—ӯпјҢеҜјиҮҙWebSocketд№ҹз«ӢеҚіе…ій—ӯ

  console.log('еҪ“еүҚзәҝзЁӢдёҚе…ій—ӯ');
}, 1000);
